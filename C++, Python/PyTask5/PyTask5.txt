import matplotlib as plt
import numpy as np
n = 10**5   # длина выборки
dim = 9     # размерность распределения

# 2. Создание вырожденной ковариационной матрицы
def CreateSingularMatrix(dim):
    mat = np.random.uniform(-1000, 1000, size=(dim,dim))
    if np.linalg.det(mat) != 0:
        mat[dim-1] = 3*mat[0]
    return mat

mat = CreateSingularMatrix(dim)
print(mat)
